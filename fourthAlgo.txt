static bool migrating = false;
static unsigned active_machines = 16; // Minimum number of active machines at start
unordered_map<CPUType_t, vector<MachineId_t>> machine_groups;



VMId_t Scheduler::FindOrCreateVM(MachineId_t machine_id, VMType_t vm_type, CPUType_t cpu_type) {
    // Check if a compatible VM already exists
    for (VMId_t vm_id : vms) {
        VMInfo_t vm_info = VM_GetInfo(vm_id);
        if (vm_info.machine_id == machine_id && vm_info.vm_type == vm_type) {
            return vm_id; // Return existing VM
        }
    }

    // Otherwise, create a new VM
    VMId_t new_vm_id = VM_Create(vm_type, cpu_type);
    VM_Attach(new_vm_id, machine_id);
    vms.push_back(new_vm_id);
    SimOutput("FindOrCreateVM(): Created new VM " + to_string(new_vm_id) + " on machine " + to_string(machine_id), 2);
    return new_vm_id;
}

// Helper function to get default VM type for a CPU type
VMType_t Scheduler::GetDefaultVMForCPU(CPUType_t cpu_type) {
    switch (cpu_type) {
        case X86:
            return LINUX;
        case POWER:
            return AIX;
        case ARM:
            return WIN;
        default:
            SimOutput("Scheduler::GetDefaultVMForCPU(): Unknown CPU type " + to_string(cpu_type), 1);
            return VMType_t(-1); // Fallback VM type
    }
}
void Scheduler::Init() {
    SimOutput("Scheduler::Init(): Initializing scheduler with machine grouping", 1);
    
    unsigned total_machines = Machine_GetTotal();
    SimOutput("Scheduler::Init(): Total number of machines is " + to_string(total_machines), 3);

    // Group machines by CPU type
    for (unsigned i = 0; i < total_machines; i++) {
        MachineInfo_t machine_info = Machine_GetInfo(MachineId_t(i));
        machine_groups[machine_info.cpu].push_back(MachineId_t(i));
    }

    // Initialize active VMs
    for (auto &[cpu_type, machines] : machine_groups) {
        unsigned init_vms = min((unsigned)(machines.size()), (unsigned) (active_machines / (machine_groups.size())));
        for (unsigned i = 0; i < init_vms; i++) {
            MachineId_t machine_id = machines[i];
            VMType_t default_vm_type = GetDefaultVMForCPU(cpu_type);
            VMId_t vm_id = FindOrCreateVM(machine_id, default_vm_type, cpu_type);
            active_machines++;
        }
    }
    SimOutput("Scheduler::Init(): Initialization complete", 1);
}

double Scheduler::CalculateMachineLoad(MachineId_t machine_id) {
    MachineInfo_t machine_info = Machine_GetInfo(machine_id);
    double total_load = 0.0;

    for (unsigned i = 0; i < machine_info.active_vms; i ++)  {
        VMInfo_t vm_info = VM_GetInfo(i);
        for (TaskId_t task_id : vm_info.active_tasks) {
            TaskInfo_t task_info = GetTaskInfo(task_id);
            total_load += static_cast<double>(task_info.total_instructions) / machine_info.performance[0]; // Base MIPS
        }
    }

    // Normalize the load to be between 0 and 1
    return std::min(1.0, total_load / machine_info.num_cpus);
}

CPUPerformance_t Scheduler::GetPStateForLoad(double load) {
    if (load > 0.75) {
        return P0; // Full performance
    } else if (load > 0.5) {
        return P1; // High performance
    } else if (load > 0.25) {
        return P2; // Medium performance
    } else {
        return P3; // Low performance
    }
}

void Scheduler::NewTask(Time_t now, TaskId_t task_id) {
    TaskInfo_t task_info = GetTaskInfo(task_id);
    SimOutput("Scheduler::NewTask(): Handling new task " + to_string(task_id), 3);

    // Step 1: Assign to an existing VM
    for (VMId_t vm : vms) {
        VMInfo_t vm_info = VM_GetInfo(vm);
        MachineInfo_t machine_info = Machine_GetInfo(vm_info.machine_id);

        if (machine_info.cpu == task_info.required_cpu && vm_info.vm_type == task_info.required_vm &&
            machine_info.memory_used + task_info.required_memory <= machine_info.memory_size) {
            VM_AddTask(vm, task_id, task_info.priority);
            SimOutput("Scheduler::NewTask(): Assigned task " + to_string(task_id) + " to VM " + to_string(vm), 2);
            return;
        }
    }

    // Step 2: Create a VM on an active machine
    for (MachineId_t machine_id : machines) {
        MachineInfo_t machine_info = Machine_GetInfo(machine_id);
        if (machine_info.cpu == task_info.required_cpu && machine_info.s_state == S0) {
            VMId_t vm_id = FindOrCreateVM(machine_id, task_info.required_vm, task_info.required_cpu);
            VM_AddTask(vm_id, task_id, task_info.priority);
            SimOutput("Scheduler::NewTask(): Created VM on active machine for task " + to_string(task_id), 2);
            return;
        }
    }

    // Step 3: Activate a new machine and create a VM
    for (auto &[cpu_type, machines] : machine_groups) {
        for (MachineId_t machine_id : machines) {
            MachineInfo_t machine_info = Machine_GetInfo(machine_id);
            if (machine_info.cpu == task_info.required_cpu && machine_info.s_state == S5) {
                Machine_SetState(machine_id, S0);
                VMId_t vm_id = FindOrCreateVM(machine_id, task_info.required_vm, task_info.required_cpu);
                VM_AddTask(vm_id, task_id, task_info.priority);
                SimOutput("Scheduler::NewTask(): Activated machine and created VM for task " + to_string(task_id), 2);
                return;
            }
        }
    }

    SimOutput("Scheduler::NewTask(): Unable to assign task " + to_string(task_id), 1);
}

void Scheduler::PeriodicCheck(Time_t now) {
    SimOutput("Scheduler::PeriodicCheck(): Performing consolidation check", 3);

    for (MachineId_t machine_id : machines) {
        MachineInfo_t machine_info = Machine_GetInfo(machine_id);

        if (machine_info.active_tasks == 0 && machine_info.s_state == S0) {
            Machine_SetState(machine_id, S5);
            SimOutput("Scheduler::PeriodicCheck(): Deactivated machine " + to_string(machine_id), 2);
        } else {
            double load = CalculateMachineLoad(machine_id);
            CPUPerformance_t p_state = GetPStateForLoad(load);
            
            Machine_SetCorePerformance(machine_id, 0, p_state); // Adjust P-state
            SimOutput("Scheduler::PeriodicCheck(): Adjusted P-state for machine " + to_string(machine_id), 2);
        }
    }
}
